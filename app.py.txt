import streamlit as st
import pandas as pd

def analyze_trends(df):
    recent = df["Crash Multiplier"].tail(10)
    avg = recent.mean()
    low_streak = (recent < 2).sum()
    high_count = (recent > 10).sum()

    if low_streak >= 4:
        prediction = "âš ï¸ Low streak detected. Possible high multiplier coming soon."
        risk = "LOW"
    elif avg > 5:
        prediction = "ğŸ”¥ High average detected. Proceed with caution."
        risk = "MEDIUM"
    elif high_count >= 2:
        prediction = "ğŸ“ˆ Multiple high crashes. Watch for drop."
        risk = "HIGH"
    else:
        prediction = "ğŸ”„ Normal pattern. No strong signals."
        risk = "MEDIUM"

    return avg, low_streak, high_count, risk, prediction
